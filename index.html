$('.datepicker').datepicker()

HTML will look something like this:

    <input type="text" class="datepicker" value="02/16/12" data-date-format="mm/dd/yy" id="dp2"/>
</div>

Make sure you are including the js file and also initializing with:

$('button[name="daterange"]').daterangepicker(
  {
    ranges: {
        'Today': [moment(), moment()],
        'Yesterday': [moment().subtract('days', 1), moment().subtract('days', 1)],
        'Last 7 Days': [moment().subtract('days', 6), moment()],
        'Last 30 Days': [moment().subtract('days', 29), moment()]
    }
  }
);
